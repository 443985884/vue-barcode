vue-barcode v0.1.0

Barcode component for Vue.js.

Examples


HTML:
<div id="app">
  <barcode value="1234567890"></barcode>
  <barcode value="1234567890" :options="{ lineColor: '#0275d8' }" tag="svg"></barcode>
  <barcode value="1234567890" tag="img"></barcode>
</div>
JavaScript:
window.addEventListener('DOMContentLoaded', function () {
  var Vue = window.Vue;
  var VueBarcode = window.VueBarcode;

  Vue.component('barcode', VueBarcode);

  new Vue({
    el: '#app',
  });
});
